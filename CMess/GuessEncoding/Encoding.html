def const_name_for(encoding)
  encoding.tr('-', '_').gsub(/\W/, '').sub(/\A\d/, 'ENC_\&').upcase
end

def get_or_set_encoding_const(encoding)
  const_defined?(const = const_name_for(encoding)) ?
    const_get(const) : set_encoding_const(encoding, const)
end
